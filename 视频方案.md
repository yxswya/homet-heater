// transcode-service.js
const Bull = require('bull');
const { exec } = require('child_process');
const fs = require('fs-extra');
const path = require('path');
const AWS = require('aws-sdk');
const { promisify } = require('util');

const execAsync = promisify(exec);

// 配置
const config = {
  redis: { host: '127.0.0.1', port: 6379 },
  videoSourceDir: '/data/uploads',
  transcodeOutputDir: '/data/transcoded',
  cdnBucket: 'my-video-cdn',
  cdnEndpoint: 'https://cdn.example.com',
  concurrency: 2,  // 同时处理2个视频（根据服务器配置）
};

// S3 配置
const s3 = new AWS.S3({
  accessKeyId: process.env.AWS_ACCESS_KEY,
  secretAccessKey: process.env.AWS_SECRET_KEY,
  region: 'us-east-1',
});

// 创建转码队列
const transcodeQueue = new Bull('transcode', {
  redis: config.redis,
  defaultJobOptions: {
    attempts: 3,
    backoff: {
      type: 'exponential',
      delay: 60000,
    },
    timeout: 7200000, // 2小时超时
  },
});

// 数据库模拟（实际用 PostgreSQL/MySQL）
const db = {
  async updateVideoStatus(videoId, status, data = {}) {
    console.log(`[DB] Video ${videoId} status: ${status}`, data);
    // await pool.query('UPDATE videos SET status = ?, ... WHERE id = ?', [...]);
  },
};

// ==========================================
// 1. 添加转码任务
// ==========================================
async function addTranscodeJob(videoId, filePath) {
  const job = await transcodeQueue.add({
    videoId,
    inputPath: filePath,
    outputDir: path.join(config.transcodeOutputDir, videoId),
    uploadedAt: new Date().toISOString(),
  }, {
    priority: 1,  // 可根据用户等级调整优先级
  });

  await db.updateVideoStatus(videoId, 'queued', {
    jobId: job.id,
  });

  return job;
}

// ==========================================
// 2. 转码处理器
// ==========================================
transcodeQueue.process(config.concurrency, async (job) => {
  const { videoId, inputPath, outputDir } = job.data;

  console.log(`[开始转码] ${videoId}`);
  await db.updateVideoStatus(videoId, 'transcoding', {
    progress: 0,
  });

  try {
    // 确保输出目录存在
    await fs.ensureDir(outputDir);

    // 执行转码
    await execAsync(
      `bash /scripts/transcode_premium.sh "${inputPath}" "${outputDir}"`,
      { maxBuffer: 1024 * 1024 * 10 } // 10MB buffer
    );

    // 上传到 CDN
    console.log(`[上传 CDN] ${videoId}`);
    await uploadToCDN(videoId, outputDir);

    // 更新数据库
    await db.updateVideoStatus(videoId, 'ready', {
      hlsUrl: `${config.cdnEndpoint}/${videoId}/master.m3u8`,
      previewUrl: `${config.cdnEndpoint}/${videoId}/preview.mp4`,
      thumbnailUrl: `${config.cdnEndpoint}/${videoId}/thumbnails/sprite.jpg`,
      completedAt: new Date().toISOString(),
    });

    console.log(`[转码完成] ${videoId}`);
    return { success: true };

  } catch (error) {
    console.error(`[转码失败] ${videoId}`, error);
    await db.updateVideoStatus(videoId, 'failed', {
      error: error.message,
    });
    throw error;
  }
});

// ==========================================
// 3. 上传到 CDN
// ==========================================
async function uploadToCDN(videoId, localDir) {
  const files = await getAllFiles(localDir);

  for (const file of files) {
    const relativePath = path.relative(localDir, file);
    const s3Key = `${videoId}/${relativePath}`;

    await s3.upload({
      Bucket: config.cdnBucket,
      Key: s3Key,
      Body: fs.createReadStream(file),
      ContentType: getContentType(file),
      CacheControl: file.endsWith('.m3u8')
        ? 'public, max-age=60'        // Playlist 短缓存
        : 'public, max-age=31536000',  // TS 文件永久缓存
    }).promise();

    console.log(`  ✓ 已上传: ${s3Key}`);
  }
}

// 递归获取所有文件
async function getAllFiles(dir) {
  const entries = await fs.readdir(dir, { withFileTypes: true });
  const files = await Promise.all(entries.map((entry) => {
    const fullPath = path.join(dir, entry.name);
    return entry.isDirectory() ? getAllFiles(fullPath) : fullPath;
  }));
  return files.flat();
}

// 根据扩展名返回 Content-Type
function getContentType(filePath) {
  const ext = path.extname(filePath).toLowerCase();
  const types = {
    '.m3u8': 'application/vnd.apple.mpegurl',
    '.ts': 'video/mp2t',
    '.mp4': 'video/mp4',
    '.jpg': 'image/jpeg',
    '.vtt': 'text/vtt',
  };
  return types[ext] || 'application/octet-stream';
}

// 5. Express API
// ==========================================
const express = require('express');
const multer = require('multer');

const app = express();
const upload = multer({ dest: config.videoSourceDir });

// 上传视频接口
app.post('/api/videos/upload', upload.single('video'), async (req, res) => {
  const videoId = generateId();  // 生成唯一 ID
  const filePath = req.file.path;

  // 重命名文件
  const newPath = path.join(config.videoSourceDir, `${videoId}.mp4`);
  await fs.rename(filePath, newPath);

  // 添加到转码队列
  const job = await addTranscodeJob(videoId, newPath);

  res.json({
    videoId,
    status: 'queued',
    jobId: job.id,
    message: '视频正在转码中，完成后将通过邮件通知您',
  });
});

// 查询转码状态
app.get('/api/videos/:videoId/status', async (req, res) => {
  const { videoId } = req.params;
  // const video = await db.getVideo(videoId);
  
  res.json({
    videoId,
    status: 'ready',  // queued | transcoding | ready | failed
    progress: 100,
    hlsUrl: `${config.cdnEndpoint}/${videoId}/master.m3u8`,
    previewUrl: `${config.cdnEndpoint}/${videoId}/preview.mp4`,
  });
});

app.listen(3000, () => {
  console.log('转码服务启动在端口 3000');
});

function generateId() {
  return Date.now().toString(36) + Math.random().toString(36).substr(2, 9);
}
